288 /**
289  * Hier wordt berekend hoe de grid/widt moet zijn van de divs.
290  * Ik heb voor deze berekening gekozen omdat:
291  * - Je een liquid div maakt en je makkelijk de breedte kan berekenen naarmaten je div's verwijdert.
292  * De breedte van de div's zijn in procenten.
293  * De berekenening: eerst wordt de wortel getrokken van het aantal foto's/divs die je hebt.
294  *                  daarna wordt dat cijfer naar boven afgerond.
295  *                  En als laatst wordt de 100 (max_width) gedeeldt door en afgronde cijfer.
296  * @return {String}: Geeft het aantal procent terug dat een div width moet hebben.
297  */
298 function setWidthDiv(){
299     return (MAX_WIDTH/(Math.ceil(Math.sqrt(numberOfImages))))+'%';
300 }

302 /**
303  * Hier wordt de gallarij gecreeërt.
304  * Er wordt een loop gemaakt voor het aantal foto's die er zijn en wordt in de container gevuld.
305  * Eerst wordt er een div gemaakt waar je de div.width in set zodat het mooi gerasterd zal worden.
306  * Dan wordt er een image gecreeërt door de functie createImage waarbij je de image meegeeft.
307  * Daarna zet je de image in de div en de div in de container.
308  * @param widthDiv : De breedte dat elk div moet hebben.
309  * @param container : De container waar je de div's (met de images erin) in zet.
310  */
311 function createGallery(widthDiv, container) {
312     for (var i = 0; i < numberOfImages; i++) {
313         var div = document.createElement('div');
314         div.style.width = widthDiv;
315         var img = createImage(album[i]);
316         div.appendChild(img);
317         container.appendChild(div);
318     }
319 }

334 /**
335  * Hier wordt een image(img) element gecreeërt nadat je als parameter een image meegeeft.
336  * Als eerst wordt er een element gecreeërt voor de foto/image.
337  * Daarna worden de waardes/attributes geset voor het element(de url wordt de src, en hij neem de id, date en photopgrapher over)
338  * Wanneer je op een image klik wordt er een event opgeroepen en daarin de functie mouseDown.
339  * Voor de functie mouseDown geeft je de event door en de image zelf(this).
340  * De reden dat ik de onmousedown hier heb bijgezet is omdat de functie alleen mag worden opgeroepen voor de image.
341  * @param image : De image die je meegeeft uit de album.
342  * @return {HTMLElement}: Geeft een img html element terug met de bijbehorende attributes.
343  */
344 function createImage(image){
345     var img = document.createElement('img');
346     img.src = image.url;
347     img.id = image.id;
348     img.alt = image.alt;
349     img.date = image.date;
350     img.photographer = image.photographer;
351     img.onmousedown= function (e){mouseDown(e, this);};
352     return img;
353 }

355 /**
356  * Set de width van alle divs die in de container zitten.
357  * Eerst wordt de numberOfImages geset, hierbij wordt het aantal div's in de container berekend en meegegeven.
358  * Daarna komt er een loop voor het aantal images/divs in de container en wordt voor elke child element de width geset.
359  * Ik heb hiervoor gekozen omdat dit uitgevoert moet worden nadat er een foto of foto's worden verwijdert.
360  * Dit is ook makkelijk te herbruiken in het geval je de raster wil veranderen.
361  *
362  * Hiervoor heb ik de javascript book 2edition geraadpleegd --> Node Relationships blz 264.
363  * @param container : De container waar alle div's zitten (met de images erin).
364  */
365 function setWidthDivs(container) {
366     numberOfImages = container.childElementCount;
367     for (var i = 0; i < numberOfImages; i++) {
368         container.childNodes.item(i).style.width = setWidthDiv();
369     }
370 }

372 /**
373  * Met deze functie kun je een image verwijderen uit de photoviewer(container).
374  * Eerst wordt de next image geset in het geval je op de spatie drukt om de image in te zoomen,
375  * de zoom gaat op deze manier meteen naar de volgende foto. De nextimage en previous (door de functie setVariablesForImages) van de image wordt als eerst geset omdat erna de image wordt verwijdert uit de container.
376  * Daarna wordt een variable aangemaakt om de div van de image die je hebt meegegeven in te zetten.
377  * Dan wordt een variable aangemaakt voor de container (deze wordt opgehaald en opgeslagen in de container).
378  * Daarna wordt de div uit de container verwijdert.
379  * De imageId wordt geset naar de nextImage.id (omdat image niet meer bestaat in de container) in het geval er op een spatie wordt gedrukt om een image in te zoomen.
380  * @param image : De image die verwijdert moet worden uit de container.
381  * @return {HTMLElement} : De container nadat de image is verwijdert.
382  */
383 function removeDivFromContainer(image) {
384     setVariablesForImages(image);
385     var div = document.getElementById(image.id).parentNode;
386     var container = document.getElementById('container');
387     container.removeChild(div);
388     imageId = nextImage.id;
389     return container;
390 }

392 /**
393  * De functie mouseDown geeft weer wat er moet gebeuren nadat je op de muis klikt.
394  * Eerst wordt de default verwijdert zodat je hier geen last van hebt in de browsers.
395  * Daarna wordt er gekeken of het event property gelijk is aan de rechter muisklik code.
396  * In het geval dit zo is wordt de image die meeggeven is verwijdert en een nieuwe container gemaakt.
397  * Dan wordt de with van alle divs geset van de container.
398  * In het geval de event property gelijk is aan de linker muisklik wordt de functie zoomIn opgeroepen en de parameter image wordt meegegeven.
399  * @param e : Het event dat wordt meegegeven.
400  * @param image : De image die wordt meegegeven.
401  */
402 function mouseDown(e, image) {
403     deleteDefault(e);
404     if (e.which === RIGHT_CLICK) {
405         var container = removeDivFromContainer(image);
406         setWidthDivs(container);
407     }
408     else if(e.which === LEFT_CLICK){
409         zoomIn(image);
410     }
411 }

557  * Dan maak je een textnode aan waar de photograaf en datum in staan.
558  * Als eerst wordt de textnode toegevoegd aan de popupDiv zodat hij bovenaan staan, dan wordt de image toegevoegt aan de popupdiv.
559  * En aan het einde wordt de popupDiv in de body gezet. De imageId wordt ook geupdate zodat je voor de overige functies die dit gebruiken doorgeeft op welke image hij gezoomed is.

570     document.body.appendChild(popupDiv);
571     imageId = image.id;
572 }

610 /**

612  * Eerst wordt er gekeken of er een div naast de div is waar de image in zit(de volgende div).
613  * Zo ja wordt de nextImage de image in de next div.
614  * Ander wordt er automatisch gekeken naar de eerste div in de container en wordt de nextImage de image in die div.
615  * @param image : De huidige image waarvan je de nextImage wil weten.
616  */
617 function setNextImage(image) {
618     if(document.getElementById(image.id).parentNode.nextSibling !== null){
619         nextImage=document.getElementById(image.id).parentNode.nextSibling.firstChild;
620     }
621     else{
622         nextImage = document.getElementById('container').firstChild.firstChild;
623     }
624 }
